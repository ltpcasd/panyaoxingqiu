```
└─────────────────────────────────────────────────────────────┘
```

**启动优化代码示例：**

```typescript
// utils/launch-optimizer.ts
export class LaunchOptimizer {
  // 预加载关键数据
  static async preloadCriticalData() {
    const promises = [
      this.preloadUserInfo(),
      this.preloadCoupleInfo(),
      this.preloadCache()
    ];
    return Promise.allSettled(promises);
  }

  // 骨架屏策略
  static renderSkeleton(page: string) {
    const skeletonMap: Record<string, string> = {
      'index': 'skeleton-index',
      'timeline': 'skeleton-timeline',
      'album': 'skeleton-album'
    };
    return skeletonMap[page] || 'skeleton-default';
  }

  // 渐进式加载
  static progressiveLoad(callbacks: Array<() => Promise<void>>) {
    callbacks.forEach((cb, index) => {
      setTimeout(() => cb(), index * 100);
    });
  }
}
```

#### 1.4.2 Redis 缓存设计

```yaml
# Redis Key 设计规范
用户相关:
  user:session:{openid}:        # 用户登录态 Hash
    fields:
      - user_id
      - token
      - expire_at
  user:info:{user_id}:          # 用户信息 String (JSON)
  user:settings:{user_id}:      # 用户设置 String (JSON)

配对相关:
  couple:info:{couple_id}:      # 配对信息 String (JSON)
  couple:users:{user_id}:       # 用户配对关系 String (couple_id)
  couple:pair_code:{code}:      # 配对码映射 String (couple_id)
  couple:online:{couple_id}:    # 在线状态 Hash

业务数据:
  timeline:{couple_id}:list     # 时光轴列表 ZSet (score: timestamp)
  album:{couple_id}:{album_id}:photos  # 相册照片 ZSet
  task:daily:{user_id}:{date}   # 每日任务 Hash
  mailbox:{couple_id}:unread    # 未读信件数量 String
  quiz:daily:{couple_id}:{date} # 今日问题 String

限流与计数:
  rate_limit:api:{path}:{user_id}  # API限流 String (计数器)
  rate_limit:sms:{phone}           # 短信限流
  
分布式锁:
  lock:couple:create:{user_id}     # 创建配对锁
  lock:task:complete:{user_id}:{task_id}  # 完成任务锁
```

### 1.5 缓存策略

#### 1.5.1 多级缓存架构

```
┌─────────────────────────────────────────────────────────────┐
│                     多级缓存架构                             │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  L1: 小程序本地缓存 (Storage)                                │
│  ├── 用户基本信息 (长期缓存)                                  │
│  ├── 配对信息 (长期缓存)                                      │
│  ├── 时光轴列表 (分页缓存，最近3页)                           │
```

```typescript
  static async getWithCache<T>(
    key: string,
    fetcher: () => Promise<T>,
    ttl: number = 300
  ): Promise<T> {
    // 1. 尝试从缓存读取
    const cached = await redis.get(key);

    }

    const data = await fetcher();
    
    // 3. 写入缓存
```

```typescript
// services/auth.service.ts

interface LoginResult {
  token: string;
  expiresIn: number;
  userInfo: UserInfo;
  coupleInfo?: CoupleInfo;
}

export class AuthService {
  // 小程序登录
  static async miniProgramLogin(code: string, encryptedData?: string, iv?: string): Promise<LoginResult> {
    // 1. 调用微信接口获取 openid 和 session_key
    const wxSession = await this.getWxSession(code);
    
    // 2. 查询或创建用户
    let user = await UserRepository.findByOpenid(wxSession.openid);
    
    if (!user) {
      // 解密用户信息（首次登录）
      const userData = encryptedData 
        ? this.decryptUserData(encryptedData, iv, wxSession.session_key)
        : null;
        
      user = await UserRepository.create({
        openid: wxSession.openid,
        unionid: wxSession.unionid,
        ...userData
      });
    }
    
    // 3. 生成 JWT Token
    const token = jwt.sign(
      { userId: user.id, openid: user.openid },
      process.env.JWT_SECRET,
      { expiresIn: '7d' }
    );
    
    // 4. 缓存登录态
    await redis.setex(
      `user:session:${user.openid}`,
      7 * 24 * 3600,
      JSON.stringify({ userId: user.id, token })
    );
    
    // 5. 查询配对信息
    const coupleInfo = await CoupleService.getCoupleInfo(user.id);
    
    return {
      token,
      expiresIn: 7 * 24 * 3600,
      userInfo: this.sanitizeUserInfo(user),
      coupleInfo
    };
  }

  // 获取微信 session
  private static async getWxSession(code: string): Promise<WxSession> {
    const url = `https://api.weixin.qq.com/sns/jscode2session`;
    const response = await axios.get(url, {
      params: {
        appid: process.env.WX_APPID,
        secret: process.env.WX_SECRET,
        js_code: code,
        grant_type: 'authorization_code'
      }
    });
    
    if (response.data.errcode) {
      throw new Error(`微信登录失败: ${response.data.errmsg}`);
    }
    
    return {
      openid: response.data.openid,
      session_key: response.data.session_key,
      unionid: response.data.unionid
    };
  }

  // 解密用户数据
  private static decryptUserData(encryptedData: string, iv: string, sessionKey: string): UserData {
    const decipher = crypto.createDecipheriv(
      'aes-128-cbc',
      Buffer.from(sessionKey, 'base64'),
      Buffer.from(iv, 'base64')
    );
    decipher.setAutoPadding(true);
    
    let decoded = decipher.update(encryptedData, 'base64', 'utf8');
    decoded += decipher.final('utf8');
    
    return JSON.parse(decoded);
  }
}
```

```typescript
  static async sendSubscribeMessage(
    openid: string,
    templateId: string,
    data: Record<string, any>,
    page?: string
  ): Promise<void> {
    const accessToken = await this.getAccessToken();
    
    const url = `https://api.weixin.qq.com/cgi-bin/message/subscribe/send?access_token=${accessToken}`;
    
    const messageData: Record<string, { value: string }> = {};
    Object.keys(data).forEach(key => {
      messageData[key] = { value: String(data[key]) };
    });
    
    const response = await axios.post(url, {
      touser: openid,
      template_id: templateId,
      page,
      data: messageData,
      miniprogram_state: 'formal', // formal/trial/developer
      lang: 'zh_CN'
    });
    
    if (response.data.errcode !== 0) {
      console.error('发送订阅消息失败:', response.data);
      throw new Error(response.data.errmsg);
    }
  }

  // 获取 Access Token
  private static async getAccessToken(): Promise<string> {
    const cacheKey = 'wx:access_token';
    let token = await redis.get(cacheKey);
    
    if (!token) {
      const url = `https://api.weixin.qq.com/cgi-bin/token`;
      const response = await axios.get(url, {
        params: {
          grant_type: 'client_credential',
          appid: process.env.WX_APPID,
          secret: process.env.WX_SECRET
        }
      });
      
      token = response.data.access_token;
      // 提前5分钟过期
      await redis.setex(cacheKey, response.data.expires_in - 300, token);
    }
    
    return token;
  }

  // 新信件通知

      receiver.openid,

        name1: letter.sender_nickname,
        thing2: letter.title || '一封新信件',
```

```typescript
        return Object.keys(obj).reduce((acc, key) => {
          acc[key] = sanitize(obj[key]);
          return acc;
        }, {} as any);
      }
      return obj;
    };
    
    req.body = sanitize(req.body);
    next();
  }
```

```typescript
// services/content-moderation.service.ts

export class ContentModerationService {
  // 文本审核
  static async checkText(content: string): Promise<ModerationResult> {
    // 敏感词过滤
    const sensitiveWords = await this.getSensitiveWords();
    for (const word of sensitiveWords) {
      if (content.includes(word)) {
        return { pass: false, reason: '包含敏感词汇', type: 'text' };
      }
    }
    
    // 调用腾讯云/阿里云文本审核API
    const result = await this.callTextModerationAPI(content);
    
    return {
      pass: result.suggestion === 'pass',
      reason: result.label,
      type: 'text',
      confidence: result.rate
    };
  }

  // 图片审核
  static async checkImage(imageUrl: string): Promise<ModerationResult> {
    const result = await this.callImageModerationAPI(imageUrl);
    
    // porn: 色情, terrorism: 暴恐, politics: 政治
    const riskTypes = ['porn', 'terrorism', 'politics', 'ad'];
    const hasRisk = riskTypes.some(type => result[type]?.suggestion !== 'pass');
    
    return {
      pass: !hasRisk,
      reason: hasRisk ? '图片包含违规内容' : undefined,
      type: 'image',
      details: result
    };
  }

  // 内容发布前审核
  static async prePublishCheck(content: {
    text?: string;
    images?: string[];
  }): Promise<boolean> {
    const checks: Promise<ModerationResult>[] = [];
    
    if (content.text) {
      checks.push(this.checkText(content.text));
    }
    
    if (content.images) {
      checks.push(...content.images.map(img => this.checkImage(img)));
    }
    
    const results = await Promise.all(checks);
    
    // 任一审核不通过则整体不通过
    const failed = results.filter(r => !r.pass);
    if (failed.length > 0) {
      throw new Error(`内容审核未通过: ${failed.map(f => f.reason).join(', ')}`);
    }
    
    return true;
  }
}
```

### 5.3 数据请求优化

#### 5.3.1 请求合并与缓存

```typescript
// utils/request.ts

class RequestManager {
  private cache: Map<string, { data: any; timestamp: number }> = new Map();
  private pendingRequests: Map<string, Promise<any>> = new Map();
  
  // 请求配置
  private config = {
    baseURL: 'https://api.panyaoxingqiu.com',
    timeout: 10000,
    retry: 2,
    retryDelay: 1000
  };

  // 带缓存的请求
  async requestWithCache<T>(
    url: string,
    options: RequestOptions = {},
    cacheDuration: number = 60000
  ): Promise<T> {
    const cacheKey = `${url}_${JSON.stringify(options.data)}`;
    
    // 检查缓存
    const cached = this.cache.get(cacheKey);
    if (cached && Date.now() - cached.timestamp < cacheDuration) {
      return cached.data;
    }
    
    // 检查是否有进行中的相同请求
    if (this.pendingRequests.has(cacheKey)) {
      return this.pendingRequests.get(cacheKey);
    }
    
    // 发起请求
    const requestPromise = this.executeRequest<T>(url, options)
      .then(data => {
        // 缓存结果
        this.cache.set(cacheKey, { data, timestamp: Date.now() });
        this.pendingRequests.delete(cacheKey);
        return data;
      })
      .catch(error => {
        this.pendingRequests.delete(cacheKey);
        throw error;
      });
    
    this.pendingRequests.set(cacheKey, requestPromise);
    return requestPromise;
  }

  // 请求聚合
  async batchRequest(requests: BatchRequestItem[]) {
    // 合并多个请求到一个接口
    return this.post('/batch', { requests });
  }

  // 接口防抖
  debounceRequest<T>(
    url: string,
    options: RequestOptions,
    wait: number = 300
  ): Promise<T> {
    return new Promise((resolve, reject) => {
      clearTimeout(this.debounceTimer);
      this.debounceTimer = setTimeout(() => {
        this.request(url, options).then(resolve).catch(reject);
      }, wait);
    });
  }

  // 网络状态自适应
  async adaptiveRequest<T>(url: string, options: RequestOptions): Promise<T> {
    const networkType = await this.getNetworkType();
    
    // 根据网络状态调整策略
    switch (networkType) {
      case 'wifi':
        // 正常请求
        return this.request(url, options);
      case '4g':
        // 压缩数据
        options.compress = true;
        return this.request(url, options);
      case '3g':
      case '2g':
        // 使用缓存，减少请求
        return this.requestWithCache(url, options, 5 * 60 * 1000);
      default:
        // 离线模式，使用本地数据
        return this.getOfflineData(url);
    }
  }
}

export const request = new RequestManager();
```

```typescript
// utils/cache-manager.ts

export class CacheManager {
  // 多级缓存
  async get<T>(key: string): Promise<T | null> {
    // 1. 内存缓存
    const memoryData = memoryCache.get(key);
    if (memoryData) return memoryData;
    
    // 2. Storage 缓存
    const storageData = wx.getStorageSync(key);
    if (storageData) {
      // 回填内存缓存
      memoryCache.set(key, storageData);
      return storageData;
    }
    
    return null;
  }

  async set(key: string, data: any, options: CacheOptions = {}) {
    const { memory = true, storage = true, expire = 3600 } = options;
    
    if (memory) {
      memoryCache.set(key, data, expire);
    }
    
    if (storage) {
      wx.setStorageSync(key, {
        data,
        expireAt: Date.now() + expire * 1000
      });
    }
  }

  // 缓存清理策略
  cleanup() {
    // 清理过期缓存
    const keys = wx.getStorageInfoSync().keys;
    keys.forEach(key => {
      const item = wx.getStorageSync(key);
      if (item && item.expireAt && item.expireAt < Date.now()) {
        wx.removeStorageSync(key);
      }
    });
  }
}
```
